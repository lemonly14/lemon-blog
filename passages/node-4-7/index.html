# 起步

const error = new Error('为什么错误')
error.errorCode = 10001
error.status = 400
error.requestUrl = `${ctx.method} + ${ctx.path}`

这样的写法实在太麻烦了，那么我们怎么解决这个问题呢？

我们使用面向对象的方式来解决这个问题。我们定义一个类，这个类继承 node.js 原来的 Error 类，然后将错误信息全部封装到类里面。

我们在 core 文件夹下新建一个文件 http-exception.js 在这个类里面使用构造函数，构造函数接收三个参数 msg，errorCode，code 他们的默认值分别为服务器异常，10000，400。 我们可以看到 HttpException 继承 node.js 原来的 Error 类。那么这样做为什么呢？

class HttpException extends Error {
  constructor(msg = '服务器异常', errorCode = 10000, code = 400) {
    super()
    this.errorCode = errorCode
    this.code = code
    this.msg = msg
  }
}

module.exports = { HttpException }

# 修改 exception

接下来我们修改一下，exception 这个中间件。

 




 












const { HttpException } = require('../core/http-exception')
const catchError = async (ctx, next) => {
  try {
    await next()
  } catch (error) {
    if (error instanceof HttpException) {
      ctx.body = {
        msg: error.msg,
        error_code: error.errorCode,
        request: `${ctx.method} + ${ctx.path}`
      }
      ctx.status = error.code
    }
  }
}

module.exports = catchError
